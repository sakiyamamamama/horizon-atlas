データベースとの接続とテーブルの作成

python

import sqlite3

# データベースファイルの名前を保持
dbname = 'TEST.db'
# データベースに接続
conn = sqlite3.connect(dbname)
# データベースを操作するカーソルオブジェクトを作成
cur = conn.cursor()
# データベースにデータをコミット
conn.commit()

# データベースの接続を切断
cur.close()
conn.close()

python

import sqlite3

# データベースファイルの名前
dbname = 'TEST.db'
# データベースに接続
conn = sqlite3.connect(dbname)

# データベースを操作するカーソルオブジェクトを作成
cur = conn.cursor()

**# テーブル作成
cur.execute(
    'CREATE TABLE person(id INTEGER PRIMARY KEY AUTOINCREMENT, name STRING, age INTEGER, gender STRING)')**

# データベースにデータをコミット
conn.commit()

# データベースの接続を切断
cur.close()
conn.close()
