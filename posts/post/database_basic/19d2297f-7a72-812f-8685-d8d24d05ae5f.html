データの削除(Delete)

sql

DELETE FROM テーブル名 WHERE 条件

python

import sqlite3

dbname = 'TEST.db'
conn = sqlite3.connect(dbname)
cur = conn.cursor()

**# データを削除する
cur.execute('DELETE FROM person WHERE id = 1')**

conn.commit()

cur.close()
conn.close()

sql

DELETE FROM person WHERE id = 1

python

cur.execute('DELETE FROM テーブル名 WHERE id = ?', (削除したい行のid,))

python

import sqlite3

dbname = 'TEST.db'
conn = sqlite3.connect(dbname)
cur = conn.cursor()

**# 任意の値を入れる
target_id = 1

# データを削除する
cur.execute('DELETE FROM person WHERE id = ?', (target_id, ))**

conn.commit()

cur.close()
conn.close()
