python

def insert_data(name, age, gender):
	dbname = 'TEST.db'
	conn = sqlite3.connect(dbname)
	cur = conn.cursor()
	
	cur.execute(**'**INSERT INTO person(name, age, gender) VALUES (?, ?, ?)', (name, age, gender))
	
	conn.commit()
	
	cur.close()
	conn.close()

python

insert_data("Jeido", 40, "Woman")

python

def get_data(id):
    dbname = 'TEST.db'
    conn = sqlite3.connect(dbname)
    cur = conn.cursor()

    cur.execute('SELECT * FROM person WHERE id = ?',(id,))
    data = cur.fetchall()

    conn.commit()

    cur.close()
    conn.close()

    return data

python

def get_all_data():
    dbname = 'TEST.db'
    conn = sqlite3.connect(dbname)
    cur = conn.cursor()

    cur.execute('SELECT * FROM person')
    data = cur.fetchall()

    conn.commit()

    cur.close()
    conn.close()

    return data

python

# 一行目のデータを取得する
print(get_data(1))

python

def update_data(name, age, gender, id):
	dbname = 'TEST.db'
	conn = sqlite3.connect(dbname)
	cur = conn.cursor()
	
	cur.execute('UPDATE person SET name = ?, age = ?, gender = ? WHERE id = ?', (name, age, gender, id))
	
	conn.commit()
	
	cur.close()
	conn.close()

python

update_data("Watanabe", 32, "Man", 1)

python

def delete_data(id):
	dbname = 'TEST.db'
	conn = sqlite3.connect(dbname)
	cur = conn.cursor()
	
	cur.execute('DELETE FROM person WHERE id = ?',(id,))
	
	conn.commit()
	
	cur.close()
	conn.close()
