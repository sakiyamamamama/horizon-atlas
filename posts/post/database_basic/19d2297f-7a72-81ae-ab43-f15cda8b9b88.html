python

import sqlite3

dbname = 'TEST.db'
conn = sqlite3.connect(dbname)
cur = conn.cursor()

**# personテーブルデータを全件取得
cur.execute('SELECT * FROM person')
data = cur.fetchall()
print(data)**

conn.commit()

cur.close()
conn.close()

import sqlite3

dbname = 'TEST.db'
conn = sqlite3.connect(dbname)
cur = conn.cursor()

**# 任意の値を入れる
target_id = 1

# personテーブルから任意の行のデータを取得
cur.execute('SELECT * FROM person WHERE id = ?',(target_id,))
data = cur.fetchall()**

# 取得したデータを出力
print(data)

conn.commit()

cur.close()
conn.close()
