データの保存(Create)

sql

INSERT INTO テーブル名(カラム名) VALUES (保存する値)

python

import sqlite3

dbname = 'TEST.db'
conn = sqlite3.connect(dbname)
cur = conn.cursor()

**# データを保存する
cur.execute('INSERT INTO person(name, age, gender) VALUES ("Sato", 15, "Man")')**

conn.commit()

cur.close()
conn.close()

cur.execute('INSERT INTO テーブル名(カラム名) VALUES (?)',(保存したい値))

python

import sqlite3

dbname = 'TEST.db'
conn = sqlite3.connect(dbname)
cur = conn.cursor()

**# 任意の値を入れる
name = "Hashimoto"
age = 26
gender = "Woman"

# データを保存する
cur.execute('INSERT INTO person(name, age, gender) VALUES (?, ?, ?)', (name, age, gender))**

conn.commit()

cur.close()
conn.close()
