データの編集(Update)

sql

UPDATE テーブル名 SET カラム名 = 値 WHERE 条件

sql

UPDATE テーブル名 SET カラム名 = 値 WHERE id = 1

python

import sqlite3

dbname = 'TEST.db'
conn = sqlite3.connect(dbname)
cur = conn.cursor()

**# データの編集
cur.execute('UPDATE person SET name = "Fujiwara", age = 69, gender = "Woman" WHERE id = 1')**

conn.commit()

cur.close()
conn.close()

sql

UPDATE person SET name = "Fujiwara", age = 69, gender = "Woman" WHERE id = 1

python

cur.execute('UPDATE テーブル名 SET カラム名 = ? WHERE id = ?', (編集後の値, 指定したいid))

python

import sqlite3

dbname = 'TEST.db'
conn = sqlite3.connect(dbname)
cur = conn.cursor()

**# 任意の値を入れる
name = "Tanaka"
age = 89
gender = "Man"
target_id = 1

# データの編集
cur.execute('UPDATE person SET name = ?, age = ?, gender = ? WHERE id = ?',(name, age, gender, target_id))**

conn.commit()

cur.close()
conn.close()
