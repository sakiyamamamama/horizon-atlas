import flet as ft

def main(page: ft.Page):
    def add_numbers(e):
        result.value = int(num1.value) + int(num2.value)
        result.update()

    num1 = ft.TextField(label="数字1")
    num2 = ft.TextField(label="数字2")
    num_row = ft.Row(controls=[num1, num2])

    add_button = ft.FilledButton(text="ADD", on_click=add_numbers)
    result = ft.Text()

    page.add(num_row, add_button, result)

ft.app(target=main)
