import flet as ft

def main(page: ft.Page):
    t = ft.Text(value="Hello, world!", color="red")
		# テキストを追加
    page.controls.append(t)
		# 画面に反映
    page.update()

ft.app(target=main)

import flet as ft

def main(page: ft.Page):
    t = ft.Text(
        value="Hello, world!",
        color="red",
        size=50,
        bgcolor="blue",
        italic=True,
    )

    page.add(t)

ft.app(target=main)

python

import flet as ft

def main(page: ft.Page):
    icon = ft.Icon(name=ft.icons.DRIVE_FILE_RENAME_OUTLINE)
    field = ft.TextField(label="名前を入力")
    button = ft.FilledButton(text="送信")

    page.add(icon, field, button)

ft.app(target=main)
