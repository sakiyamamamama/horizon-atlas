import flet as ft

def main(page: ft.Page):
    datepicker = ft.DatePicker(
    )

    row1 = ft.Row(
        controls=[
            ft.Icon(name=ft.icons.DRIVE_FILE_RENAME_OUTLINE, size=40),
            ft.TextField(label="名前", height=55, expand=1),
            ft.FilledButton(text="送信", height=40),
        ]
    )
    row2 = ft.RadioGroup(
        content=ft.Row(
            controls=[
                ft.Radio(value="first", label="１回生", expand=1),
                ft.Radio(value="second", label="２回生", expand=1),
                ft.Radio(value="third", label="３回生", expand=1),
                ft.Radio(value="fourth", label="４回生", expand=1),
            ]
        )
    )
    row3 = ft.Row(
        controls=[
            ft.Text(value="学部", width=50, text_align="CENTER"),
            ft.Dropdown(
                options=[
                    ft.dropdown.Option(text="先端理工学部"),
                    ft.dropdown.Option(text="農学部"),
                    ft.dropdown.Option(text="社会学部"),
                ],
                height=55,
                expand=1,
            )
        ]
    )
    row4 = ft.Row(
        controls=[
            ft.Text(value="誕生日", width=50, text_align="CENTER"),
            ft.Dropdown(
                options=[ft.dropdown.Option(option) for option in range(2024,1900,-1)],
                label="年",
                expand=3,
            ),
            ft.Dropdown(
                options=[ft.dropdown.Option(option) for option in range(1,13)],
                label="月",
                expand=2,
            ),
            ft.Dropdown(
                options=[ft.dropdown.Option(option) for option in range(1,32)],
                label="日",
                expand=2,
            ),
        ]
    )
    row5 = ft.Row(
        controls=[
            ft.Text(value="キッカケ", expand=1),
            ft.Checkbox(label="Twitter", value=False, expand=1),
            ft.Checkbox(label="Instagram", value=False, expand=1),
            ft.Checkbox(label="ビラ", value=False, expand=1),
            ft.Checkbox(label="その他", value=False, expand=1),
        ]
    )
    
    column = ft.Column(controls=[row1, row2, row3, row4, row5])

    page.add(column)

ft.app(target=main)
