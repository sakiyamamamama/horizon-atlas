import flet as ft

def main(page: ft.Page):

    def route_change(route):
        page.views.append(
            ft.View(
                "/",
                [
                    ft.Text("Here is HomePage"),
                    ft.ElevatedButton("Visit NEXT_PAGE", on_click=lambda _: page.go("/next")),
                ],
            )
        )
        if page.route == "/next":
            page.views.append(
                ft.View(
                    "/next",
                    [
                        ft.Text("Here is NextPage"),
                        ft.ElevatedButton("Go Home", on_click=lambda _: page.go("/")),
                    ],
                )
            )
        page.update()

    page.on_route_change = route_change
    page.go(page.route)


ft.app(main)

import flet as ft

def main(page: ft.Page):
    def route_change(e):

        template_route = ft.TemplateRoute(page.route)

        if template_route.match("/user/:id"):
            user_id = template_route.id  # 動的パラメータを取得
            page.views.append(
                ft.View(
                    f"/user/{user_id}",
                    controls=[
                        ft.Text(f"ユーザーID: {user_id}のページです"),
                        ft.ElevatedButton("ホームに戻る", on_click=lambda _: page.go("/"))
                    ]
                )
            )
        elif template_route.match("/"):
            user_id_field = ft.TextField(label="ユーザーIDを入力してください")
            # ホームページの表示
            page.views.append(
                ft.View(
                    "/",
                    controls=[
                        ft.Text("ホームページ"),
                        user_id_field,
                        ft.ElevatedButton("ユーザーページへ", on_click=lambda _: page.go(f"/user/{user_id_field.value}"))
                    ]
                )
            )
        
        page.update()

    page.on_route_change = route_change
    page.go(page.route)

ft.app(target=main)
