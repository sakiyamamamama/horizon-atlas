12. クラス (1)

python

class Student:
	　# コンストラクタ(インスタンスを作成する際に呼ばれる特別なメソッド)
    def __init__(self, name, score):
		    # 属性
        self.name = name
        self.score = score
    
    # displayメソッドを定義（生徒の名前と点数を表示する）
    def display(self):
        print("名前:", self.name, "\n点数:", self.score)
