3. 変数の型 (1)

python

a = 10   変数 a に10を代入
b = 5    変数 b に5を代入
a と b を出力
print(a, b)

python

a = 10   #変数 a に10を代入
b = 5    #変数 b に5を代入
#a と b を出力
print(a, b)

python

a = 10    #a は整数型
b = 3.5   #b は実数型
c = "str" #c は文字列型

python

a = 5      #この時点では a は整数型
a = 4.2    #実数が代入されたため、a は実数型に変更
a = "char" #文字列が代入されたため、a は文字列型に変更

python


a = 10   
b = 3.5
c = "str"
print(type(a))  #a の型を確認
print(type(b))  #b の型を確認
print(type(c))  #c の型を確認
