7. リストと辞書 (1)

python

#変数listを[1, 2, 3]を格納するリストとする
list = [1, 2, 3]
print(list)       #listを丸ごと出力

text

＜出力結果＞
[1, 2, 3]

python

list = [1, 2, 3]
print(list[0])
print(list[-1])

text

＜出力結果＞
1
3

python

list = [0, 1, 2, 3, 4, 5, 6]
sliced_list = list[2:5]
print(sliced_list)

text

＜出力結果＞
[2, 3, 4]

python

list = [1, 2, 3, 4, 5]
list.append(6)     #listに新しい要素6を追加
print(list)

text

＜出力結果＞
[1, 2, 3, 4, 5, 6]

python

my_dict = {"apple": 100, "banana": 200, "orange": 300}

text

＜出力結果＞
{'apple': 100, 'banana': 200, 'orange': 300}

python

my_dict = {"apple": 100, "banana": 200, "orange": 300}
print(my_dict["apple"])

text

＜出力結果＞
100

python

my_dict = {"apple": 100, "banana": 200, "orange": 300}
my_dict["grape"] = 400
print(my_dict)

text

＜出力結果＞
{'apple': 100, 'banana': 200, 'orange': 300, 'grape': 400}
