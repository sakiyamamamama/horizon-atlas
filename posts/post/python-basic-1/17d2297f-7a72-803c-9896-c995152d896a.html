def 関数の名前(引数を受け取る変数):
    関数を呼び出したときに行う処理
    return 戻り値として返したい値

python

def find_triangle_area(tei, taka):  #受け取った底辺と高さを変数 tei と taka とする
    area = tei * taka / 2
    return area     #変数 area を戻り値として返す

python

def find_triangle_area(tei, taka):
    area = tei * taka / 2
    return area

triangle_area = find_triangle_area(5, 3)
print(triangle_area)
