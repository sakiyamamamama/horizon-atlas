4. 四則演算 (1)

python

a = 10 + 5  #足し算
b = 10 - 5  #引き算
c = 10 * 5  #掛け算
d = 10 / 5  #割り算
print("10 + 5 =", a)
print("10 - 5 =", b)
print("10 * 5 =", c)
print("10 / 5 =", d)

python

v1 = 10
v2 = 5
a = v1 + v2  #足し算
b = v1 - v2  #引き算
c = v1 * v2  #掛け算
d = v1 / v2  #割り算
print("v1 + v2 =", a)
print("v1 - v2 =", b)
print("v1 * v2 =", c)
print("v1 / v2 =", d)

python

v1 = 5
v2 = 2
a = v1 % v2     #割り算のあまり
b = v1 ** v2    #累乗
c = v1 // v2    #切り捨ての割り算
print("v1 % v2 =", a)
print("v1 ** v2 =", b)
print("v1 // v2 =", c)
