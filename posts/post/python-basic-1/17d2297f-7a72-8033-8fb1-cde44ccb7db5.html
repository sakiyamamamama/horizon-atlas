python

a = "5"     #文字列として 5 を代入
print(type(a))  #a の型を確認
a = int(a)      #a を整数型に変換し、a に再代入
print(type(a))  #再度 a の型を確認

python

print("三角形の面積を求めます")

b = input("底辺を入力：")

b = int(b)
print("高さ", a, "、底辺", b, "の三角形の面積は", a * b / 2)
