クラスの継承

クラスの継承

python

class Enemy:
    def __init__(self, name, hp, attack):
		    # 属性
        self.name = name 
        self.hp = hp      
        self.attack = attack  

    def take_damage(self, damage):
        # ダメージを受ける
        self.hp -= damage
        print(f"{self.name} took {damage} damage! HP is now {self.hp}.")
        if self.hp <= 0:
            self.die()

    def die(self):
        # 敵が倒された時
        print(f"{self.name} has been defeated!")

python

class Slime(Enemy):
	def __init__(self, name="slime", hp=10, attack=1):
			super().__init__(name, hp, attack)
				
	# 独自のメソッド（分裂）
	def split(self):
			print(f"{self.name} split into two smaller slimes!")

python

# インスタンス作成
slime1 = Slime()

# スライムのステータスを出力
print(f"Name: {slime1.name}, HP: {slime1.hp}, Attack: {slime1.attack}")

# スライムがダメージを受ける
slime1.take_damage(4)

# スライムが分裂する
slime1.split()
